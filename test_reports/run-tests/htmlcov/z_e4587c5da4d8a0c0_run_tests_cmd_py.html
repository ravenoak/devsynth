38def _parse_feature_options(values: list[str]) -> dict[str, bool]: 

39 """Convert ``--feature`` options into a dictionary. 

40 

41 Each value should be in the form ``name`` or ``name=value`` where ``value`` 

42 can be interpreted as a boolean. If ``value`` is omitted it defaults to 

43 ``True``. 

44 """ 

45 

46 # Typer always passes a list for multi-option flags; guard defensively 

47 # for runtime safety. 

48 if not values: 

49 return {} 

50 

51 result: dict[str, bool] = {} 

52 for item in values: 

53 if "=" in item: 

54 name, raw = item.split("=", 1) 

55 result[name] = raw.lower() not in {"0", "false", "no"} 

56 else: 

57 result[item] = True 

58 return result

109def _parse_pytest_addopts(addopts: str | None) -> list[str]: 

110 """Parse ``PYTEST_ADDOPTS`` into discrete tokens.""" 

111 

112 if not addopts or not addopts.strip(): 

113 return [] 

114 try: 

115 return shlex.split(addopts) 

116 except ValueError: 

117 # Fall back to a naive split when quoting is imbalanced; pytest will 

118 # still receive the original string and surface the parsing error. 

119 return addopts.split()

122def _addopts_has_plugin(tokens: list[str], plugin: str) -> bool: 

123 """Check whether a ``-p`` plugin directive exists for the given plugin.""" 

124 

125 for index, token in enumerate(tokens): 

126 if token == "-p" and index + 1 < len(tokens) and tokens[index + 1] == plugin: 

127 return True 

128 if token.startswith("-p") and token[2:] == plugin: 

129 return True 

130 return False

133def _coverage_instrumentation_disabled(tokens: list[str]) -> bool: 

134 """Return ``True`` when coverage instrumentation is explicitly disabled.""" 

135 

136 if "--no-cov" in tokens: 

137 return True 

138 if _addopts_has_plugin(tokens, "no:cov"): 

139 return True 

140 if _addopts_has_plugin(tokens, "no:pytest_cov"): 

141 return True 

142 return False

145def _coverage_instrumentation_status() -> tuple[bool, str | None]: 

146 """Determine whether pytest-cov instrumentation is active.""" 

147 

148 addopts = os.environ.get("PYTEST_ADDOPTS", "") 

149 tokens = _parse_pytest_addopts(addopts) 

150 if "--no-cov" in tokens: 

151 return False, "PYTEST_ADDOPTS contains --no-cov" 

152 if _addopts_has_plugin(tokens, "no:cov"): 

153 return False, "PYTEST_ADDOPTS disables pytest-cov via -p no:cov" 

154 if _addopts_has_plugin(tokens, "no:pytest_cov"): 

155 return False, "PYTEST_ADDOPTS disables pytest-cov via -p no:pytest_cov" 

156 if os.environ.get( 

157 "PYTEST_DISABLE_PLUGIN_AUTOLOAD" 

158 ) == "1" and not _addopts_has_plugin(tokens, "pytest_cov"): 

159 return False, "pytest plugin autoload disabled without -p pytest_cov" 

160 return True, None
